create table user
(
		id 				int auto_increment comment 'primary key'
				primary key,
		no 				varchar(20)							null comment 'account',
		name			varchar(100)						not null comment 'name',
		password  varchar(20)							not null comment 'password',
		age 			int 										null,
		sex       int 										null comment 'sex',
		phone			varchar(20)  						null comment 'phone',
		role_id		int 										null comment 'role 0 - super admin, 1 - admin, 2 - normal user',
		isValid		varchar(4) default 'Y' 	null comment 'is valid, Y yes, other invalid'

)
		charset = utf8;

---


CREATE TABLE `menu` (

  `menuComponent` varchar(200) DEFAULT NULL,
  `menuIcon` varchar(100) DEFAULT NULL,
  PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;
